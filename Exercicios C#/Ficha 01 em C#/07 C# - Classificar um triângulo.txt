using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Ficha01Q07
{
    class Program
    {
        static void Main(string[] args)
        {
            // 07 C# - Classificar um triângulo
            int a, b, c;

            Console.WriteLine("Insira os três lados de um triângulo: ");
            a = Convert.ToInt16(Console.ReadLine());
            b = Convert.ToInt16(Console.ReadLine());
            c = Convert.ToInt16(Console.ReadLine());

            if (a==b && b==c)
                Console.Write(":: Triângulo equilátero\n\n");
            else if (a==b||a==c||b==c)
                Console.Write(":: Triângulo isósceles\n\n");
            else
                Console.Write(":: Triângulo escaleno\n\n");

           // Console.Read();
        }
    }
}
