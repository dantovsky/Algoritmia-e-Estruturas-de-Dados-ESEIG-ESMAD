using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Ficha01Q09
{
    class Program
    {
        static void Main(string[] args)
        {
            // 06 C# - Saber se é par ou ímpar
            int num;

            Console.Write("Númreo que deseja saber se é par ou ímpar: ");
            num = Convert.ToInt16(Console.ReadLine());

            if (num % 2 == 0)
                Console.Write("O número {0} é par.", num);
            else
                Console.Write("O número {0} é ímpar.", num);

            Console.Read();
        }
    }
}
