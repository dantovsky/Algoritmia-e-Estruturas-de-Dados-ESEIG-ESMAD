inteiro codigo_planeta
texto nome_planeta

                nome_planeta <- "Mercúrio"

                nome_planeta <- "Vénus"
                
    caso 3 : novo_peso <- peso * 0.38 
                 nome_planeta <- "Marte"
                 
    caso 4 : novo_peso <- peso * 2.64
                 nome_planeta <- "Júpiter"
                
    caso 5 : novo_peso <- peso * 1.15
                nome_planeta <- "Saturno"
                
    caso 6 : novo_peso <- peso * 1.17
                nome_planeta <- "Urano"
fim escolhe

escrever "O teu peso em ", nome_planeta, " é ", novo_peso, " quilos."
