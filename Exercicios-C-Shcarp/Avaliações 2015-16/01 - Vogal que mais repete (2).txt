using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication1
{
    class Program
    {
        public static int contaVogal(char vogal, string texto)
        {
            int contador = 0;
            for (int i = 0; i < texto.Length; i++)
            {
                if (Char.ToLower(texto[i]) == vogal)
                {
                    contador++;
                }
            }
            return contador;
        }
        static void Main(string[] args)
        {
            // Teste 2 - AED 2015 | 01 - Vogal que mais repete
            // Linguagem C#
            Console.Write("Texto: ");
            string texto = Console.ReadLine();
            int maiorRepeticao = 0;
            int contTemp = 0;
            char vogalMaisRepetida = ' ';

            foreach (char vogal in texto)
            {
                switch (vogal)
                {
                    case 'a':
                        contTemp = contaVogal(vogal, texto);
                        if (contTemp > maiorRepeticao)
                        {
                            maiorRepeticao = contTemp;
                            vogalMaisRepetida = vogal;
                        }
                        break;
                    case 'e':
                        contTemp = contaVogal(vogal, texto);
                        if (contTemp > maiorRepeticao)
                        {
                            maiorRepeticao = contTemp;
                            vogalMaisRepetida = vogal;
                        }
                        break;
                    case 'i':
                        contTemp = contaVogal(vogal, texto);
                        if (contTemp > maiorRepeticao)
                        {
                            maiorRepeticao = contTemp;
                            vogalMaisRepetida = vogal;
                        }
                        break;
                    case 'o':
                        contTemp = contaVogal(vogal, texto);
                        if (contTemp > maiorRepeticao)
                        {
                            maiorRepeticao = contTemp;
                            vogalMaisRepetida = vogal;
                        }
                        break;
                    case 'u':
                        contTemp = contaVogal(vogal, texto);
                        if (contTemp > maiorRepeticao)
                        {
                            maiorRepeticao = contTemp;
                            vogalMaisRepetida = vogal;
                        }
                        break;
                    default:
                        break;
                }
            }
            // Apresentação do resultado final
            Console.Write("Vogal que mais repete: {0}\nNº repetições: {1}", vogalMaisRepetida, maiorRepeticao);
        }
    }
}
